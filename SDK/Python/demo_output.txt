使用示例:
from stardust_sdk import StardustClient

client = StardustClient(
    server="http://star.example.com:6600",
    app_id="MyApp",
    secret="MySecret",
    scope="prod"
)
client.start()

# APM 监控
with client.new_span("operation") as span:
    span.tag = "info"
    do_something()

# 配置中心
db_host = client.get_config("database.host", "localhost")
client.on_config_change(lambda cfg: print(f"Config updated: {cfg}"))

client.stop()
